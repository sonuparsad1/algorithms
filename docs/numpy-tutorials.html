ðŸ’¡ Quick Start

import numpy as np

# Create an array
arr = np.array([1, 2, 3, 4, 5])

# Index and slice
print(arr[0])      # First element
print(arr[1:3])    # Elements 1 and 2

# Create with functions
zeros = np.zeros(5)
ones = np.ones(5)
arange = np.arange(0, 10, 2)
